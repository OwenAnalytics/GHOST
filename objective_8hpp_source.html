247  {
248  int best = 0;

254  {
255  if( vecGlobalCosts[i] == bestEstimatedCost
256  && vecGlobalCosts[i] < numeric_limits<int>::max()
257  && heuristicValueHelper.at( i ) == bestHelp )
258  {
259  draw.push_back(i);
260  }
261  else
262  if( vecGlobalCosts[i] < bestEstimatedCost
263  || ( vecGlobalCosts[i] == bestEstimatedCost
264  && vecGlobalCosts[i] < numeric_limits<int>::max()
265  && heuristicValueHelper.at( i ) < bestHelp ) )
266  {
267  draw.clear();
268  bestEstimatedCost = vecGlobalCosts[i];
269 
270  // WARNING! This line implies we start our domain by the value -1
271  // which is not necessarily the case. TOFIX
272  bestValue = i - 1;
273  if( heuristicValueHelper.at( i ) < bestHelp )
274  bestHelp = heuristicValueHelper.at( i );
275  best = i;
276  }
277  }
278 
279  if( draw.size() > 1 )
280  {
281  int i = draw[ const_cast<Objective*>(this)->randomVar.getRandNum( draw.size() ) ];
