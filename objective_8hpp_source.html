256  virtual int v_heuristicValue( const std::vector< double > &vecGlobalCosts,
257  double &bestEstimatedCost,
258  int &bestValue ) const
259  {
260  int best = 0;
261  double bestHelp = numeric_limits<int>::max();
262 
263  vector<int> draw;
264 
265  for( int i = 0; i < vecGlobalCosts.size(); ++i )
266  {
267  if( vecGlobalCosts[i] == bestEstimatedCost
268  && vecGlobalCosts[i] < numeric_limits<int>::max()
269  && heuristicValueHelper.at( i ) == bestHelp )
270  {
271  draw.push_back(i);
272  }
273  else
274  if( vecGlobalCosts[i] < bestEstimatedCost
275  || ( vecGlobalCosts[i] == bestEstimatedCost
276  && vecGlobalCosts[i] < numeric_limits<int>::max()
277  && heuristicValueHelper.at( i ) < bestHelp ) )
278  {
279  draw.clear();
280  bestEstimatedCost = vecGlobalCosts[i];
281  bestValue = i - 1;
282  if( heuristicValueHelper.at( i ) < bestHelp )
283  bestHelp = heuristicValueHelper.at( i );
284  best = i;
285  }
286  }
287 
288  if( draw.size() > 1 )
289  {
290  int i = draw[ const_cast<Objective*>(this)->randomVar.getRandNum( draw.size() ) ];
291 
292  bestEstimatedCost = vecGlobalCosts[i];
293  bestValue = i - 1;
294  if( heuristicValueHelper.at( i ) < bestHelp )
295  bestHelp = heuristicValueHelper.at( i );
296  best = i;
297  }
298 
299  return best;
300  }
