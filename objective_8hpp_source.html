241  double &bestEstimatedCost,
242  int &bestValue ) const
243  {
244  int best = 0;
245  double bestHelp = numeric_limits<int>::max();
246 
247  vector<int> draw;
248 
249  for( int i = 0; i < vecGlobalCosts.size(); ++i )
250  {
251  if( vecGlobalCosts[i] == bestEstimatedCost
252  && vecGlobalCosts[i] < numeric_limits<int>::max()
253  && heuristicValueHelper.at( i ) == bestHelp )
254  {
255  draw.push_back(i);
256  }
257  else
258  if( vecGlobalCosts[i] < bestEstimatedCost
259  || ( vecGlobalCosts[i] == bestEstimatedCost
260  && vecGlobalCosts[i] < numeric_limits<int>::max()
261  && heuristicValueHelper.at( i ) < bestHelp ) )
262  {
263  draw.clear();
264  bestEstimatedCost = vecGlobalCosts[i];
265 
266  // WARNING! This line implies we start our domain by the value -1
267  // which is not necessary the case. TOFIX
268  bestValue = i - 1;
269  if( heuristicValueHelper.at( i ) < bestHelp )
270  bestHelp = heuristicValueHelper.at( i );
271  best = i;
272  }
273  }
274 
275  if( draw.size() > 1 )
276  {
277  int i = draw[ const_cast<Objective*>(this)->randomVar.getRandNum( draw.size() ) ];
