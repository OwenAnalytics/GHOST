263  template <typename TypeVariable>
264  int Objective<TypeVariable>::expert_heuristic_value( vector< TypeVariable > *variables,
265  TypeVariable *var,
266  const vector< int >& possible_values ) const
267  {
268  double minCost = numeric_limits<double>::max();
269  double simulatedCost;
270 
271  int backup = var->get_value();
272  vector<int> bestValues;
273 
274  for( auto& v : possible_values )
275  {
276  var->set_value( v );
277  simulatedCost = cost( variables );
278 
279  if( minCost > simulatedCost )
280  {
281  minCost = simulatedCost;
282  bestValues.clear();
283  bestValues.push_back( v );
284  }
285  else
286  if( minCost == simulatedCost )
287  bestValues.push_back( v );
288  }
289 
290  var->set_value( backup );
291 
292  return bestValues[ random.get_random_number( bestValues.size() ) ];
293  }
