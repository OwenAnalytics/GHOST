238  template <typename TypeVariable>
239  int Objective<TypeVariable>::expert_heuristic_value( vector< TypeVariable > *variables,
240  TypeVariable *var,
241  const vector< int >& possible_values ) const
242  {
243  double minCost = numeric_limits<double>::max();
244  double simulatedCost;
245 
246  int backup = var->get_value();
247  vector<int> bestValues;
248 
249  for( auto& v : possible_values )
250  {
251  var->set_value( v );
252  simulatedCost = cost( variables );
253 
254  if( minCost > simulatedCost )
255  {
256  minCost = simulatedCost;
257  bestValues.clear();
258  bestValues.push_back( v );
259  }
260  else
261  if( minCost == simulatedCost )
262  bestValues.push_back( v );
263  }
264 
265  var->set_value( backup );
266 
267  return bestValues[ random.get_random_number( bestValues.size() ) ];
268  }
