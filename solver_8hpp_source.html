630  template <typename TypeVariable, typename TypeConstraint>
631  double Solver<TypeVariable, TypeConstraint>::simulate_permutation_cost( TypeVariable* worstVariable,
632  TypeVariable& otherVariable,
633  const vector<double>& costConstraints,
634  double currentSatCost ) const
635  {
636  double newCurrentSatCost = currentSatCost;
637  int tmp = worstVariable->get_value();
638  worstVariable->set_value( otherVariable.get_value() );
639  otherVariable.set_value( tmp );
640 
641  vector<bool> compted( costConstraints.size(), false );
642 
643  for( auto& c : _mapVarCtr[ *worstVariable ] )
644  {
645  newCurrentSatCost += ( c->cost() - costConstraints[ c->get_id() - _ctrOffset ] );
646  compted[ c->get_id() - _ctrOffset ] = true;
647  }
648 
649  for( auto& c : _mapVarCtr[ otherVariable ] )
650  if( !compted[ c->get_id() - _ctrOffset ] )
651  newCurrentSatCost += ( c->cost() - costConstraints[ c->get_id() - _ctrOffset ] );
652 
653  // We must roll back to the previous state before returning the new cost value.
654  tmp = worstVariable->get_value();
655  worstVariable->set_value( otherVariable.get_value() );
656  otherVariable.set_value( tmp );
657 
658  return newCurrentSatCost;
659  }
